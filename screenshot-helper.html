üîß Automated Screenshot Script

Want to automate this? Here's a Node.js script:

const puppeteer = require('puppeteer'); async function takeScreenshots() { const browser = await puppeteer.launch(); const page = await browser.newPage(); await page.setViewport({ width: 1200, height: 800 }); // Open landing page await page.goto('file://' + __dirname + '/index.html'); // Hero section await page.screenshot({ path: 'landing-hero.png', clip: { x: 0, y: 0, width: 1200, height: 600 } }); // Features section await page.evaluate(() => { document.querySelector('.features').scrollIntoView(); }); await page.waitForTimeout(500); await page.screenshot({ path: 'features-section.png', clip: { x: 0, y: 200, width: 1200, height: 400 } }); // How it works await page.evaluate(() => { document.querySelector('.how-it-works').scrollIntoView(); }); await page.waitForTimeout(500); await page.screenshot({ path: 'how-it-works.png', clip: { x: 0, y: 200, width: 1200, height: 400 } }); // Testimonials await page.evaluate(() => { document.querySelector('.testimonials').scrollIntoView(); }); await page.waitForTimeout(500); await page.screenshot({ path: 'testimonials.png', clip: { x: 0, y: 200, width: 1200, height: 400 } }); await browser.close(); console.log('‚úÖ Screenshots taken!'); } takeScreenshots();

To use:

npm install puppeteer node screenshot-script.js
